export const ModalTemplate = ({
  id = 'sample-id',
  buttonTitle = "Open Modal",
  content = <p>Modal Content</p>,
  title = "Modal title",
  isAlertModal = false,
  size = 'medium'
}) => {
  const [show, setShow] = useState(false);
  const buttonRef = useRef(null);
  const hideModal = () => setShow(false);
  return (
    <div>
      <Button ref={buttonRef} onClick={() => setShow(true)}>
        {buttonTitle}
      </Button>
      <Modal
        id={id}
        show={show}
        title={title}
        isAlertModal={isAlertModal}
        onHide={hideModal}
        size={size}
        triggerElement={buttonRef.current}
        classNames={{ content: "monday-storybook-modal-new-content-wrapper" }}
      >
        {content}
        <div className="monday-storybook-modal-new-buttons">
          <Button kind={Button.kinds.SECONDARY} onClick={hideModal}>
            Cancel
          </Button>
          <Button kind={Button.kinds.PRIMARY} onClick={hideModal}>
            Confirm
          </Button>
        </div>
      </Modal>
    </div>
  );
};
