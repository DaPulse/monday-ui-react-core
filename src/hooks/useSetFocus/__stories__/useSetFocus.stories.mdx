## Overview

Hook for controlling focus on specific component e.g. Input.

<Canvas>
  <Story name="Overview">
    {() => {
      const focusCallback = () => {};
      const blurCallback = () => {};
      const ref = useRef(null);
      const { isFocused, focus, blur } = useSetFocus({ ref, focusCallback, blurCallback });
      return (
        <Flex direction={Flex.directions.COLUMN}>
          <InputField ref={ref} placeholder="Input..." />
          <Button onClick={focus} className="use-set-focus-control-button">
            Focus
          </Button>
          <Button onClick={blur} className="use-set-focus-control-button">
            Unfocus
          </Button>
          <div className="use-set-focus-label">
            Is focused: {isFocused?.toString()}
          </div>
        </Flex>
      );
    }}
  </Story>
