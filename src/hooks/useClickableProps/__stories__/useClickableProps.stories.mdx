## Overview

Return props for adding clickable functionality to the element except for the styles and classNames.

<Canvas>
  <Story name="Overview">
    {() => {
      const ref = useRef(null);
      const onClickCallback = useCallback(() => alert("click!"), []);
      const clickableProps = useClickableProps(
        {
          onClick: onClickCallback,
          id: "clickable-id",
          ariaHidden: false,
          ariaHasPopup: false,
          ariaExpanded: false
        },
        ref
      );
      return (
        <div {...clickableProps} className="monday-storybook-use-clickable-props">
          I act like a button
        </div>
      );
    }}
  </Story>
</Canvas>

<Tip title="Alternative">
  Most of the times you might want to use
  <Link size={Link.sizes.SMALL} href="/?path=/docs/accessibility-clickable--overview">
    Clickable component
  </Link>
  to apply the styles as well. Use this hook when you want to customize styles by yourself.
</Tip>
